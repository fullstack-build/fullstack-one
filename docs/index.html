Getting Started

1. Define your model

First things first, we need to define our first model. 

Head therefor to /entities and create a new File Task.ts and add the following Code:

import { BaseEntity, Entity, Column, PrimaryGeneratedColumn, CreateDateColumn, UpdateDateColumn } from "@soniq/db";

@Entity()
export default class Task extends BaseEntity {
  @PrimaryGeneratedColumn()
  public id: string;

  @CreateDateColumn()
  public readonly createdAt!: string;

  @UpdateDateColumn()
  public readonly updatedAt!: string;

  @Column({ gqlType: "String", type: "character varying" })
  public title: string;
}
